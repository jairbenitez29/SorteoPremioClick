// AUTO-LOGIN: Escuchar cuando el usuario se autentica automáticamente desde la app móvil
window.addEventListener('userAuthenticated', function(event) {
    const { user, token } = event.detail;
    console.log('✅ Usuario autenticado automáticamente desde la app:', user);
    
    // Actualizar el estado de autenticación en tu aplicación
    // Esto debería llamar a la misma función que usas cuando el usuario hace login manual
    if (typeof updateAuthState === 'function') {
        updateAuthState(user, token);
    } else if (typeof setUser === 'function') {
        setUser(user);
    } else {
        // Si no tienes una función específica, actualiza manualmente:
        // Guardar usuario en una variable global si la usas
        window.currentUser = user;
        
        // Actualizar la UI
        updateUserUI(user);
        
        // Autenticar el socket si lo usas
        if (window.socket && token) {
            window.socket.emit('authenticate', { token: token });
        }
    }
});

// Verificar sesión existente al cargar la página
document.addEventListener('DOMContentLoaded', function() {
    const token = localStorage.getItem('token');
    const userStr = localStorage.getItem('user');
    
    if (token && userStr) {
        try {
            const user = JSON.parse(userStr);
            console.log('👤 Sesión encontrada al cargar:', user);
            
            // Actualizar el estado de autenticación
            if (typeof updateAuthState === 'function') {
                updateAuthState(user, token);
            } else if (typeof setUser === 'function') {
                setUser(user);
            } else {
                // Actualizar manualmente
                window.currentUser = user;
                updateUserUI(user);
                
                // Autenticar socket
                if (window.socket && token) {
                    window.socket.emit('authenticate', { token: token });
                }
            }
        } catch (error) {
            console.error('Error al cargar sesión:', error);
            localStorage.removeItem('token');
            localStorage.removeItem('user');
        }
    }
});

// Función helper para actualizar la UI del usuario (ajusta según tu código)
function updateUserUI(user) {
    if (!user) return;
    
    // Mostrar información del usuario
    const userInfo = document.getElementById('userInfo');
    const userNameDisplay = document.getElementById('userNameDisplay');
    const loginBtn = document.getElementById('loginBtn');
    
    if (userInfo && userNameDisplay) {
        userNameDisplay.textContent = user.nombre || user.email || 'Usuario';
        userInfo.style.display = 'flex';
    }
    
    if (loginBtn) {
        loginBtn.style.display = 'none';
    }
    
    // Mostrar botón de ganadores si es admin
    const ganadoresBtn = document.getElementById('ganadoresBtn');
    if (ganadoresBtn && user.rol === 'admin') {
        ganadoresBtn.style.display = 'inline-block';
    }
    
    // Cerrar modales de login/registro si están abiertos
    const loginModal = document.getElementById('loginModal');
    const registerModal = document.getElementById('registerModal');
    if (loginModal) loginModal.style.display = 'none';
    if (registerModal) registerModal.style.display = 'none';
}
